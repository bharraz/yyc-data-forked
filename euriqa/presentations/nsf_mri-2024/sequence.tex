    \pgfoothis.drawTweezer(#1,#2,pyplotc4);
  }

    \fill [#4,path fading=glow2 fading] (#1,#2) circle (#3);
  }

    \pgfoothis.drawAtom(#1,#2,#3,pyplotc0);
  }
  \method drawUpAtom(#1,#2,#3) {
    \pgfoothis.drawAtom(#1,#2,#3,pyplotc1);
  }
}
